def avg_speed_by_borough(df):
  """
  Calculates the average speed per borough in New York.

  Args:
    df: A pandas DataFrame containing bus speed data with a 'Borough' column
        and a column representing speed (e.g., 'Speed').

  Returns:
    A pandas Series with the average speed for each borough.
  """
  # Assuming 'Speed' is the column with speed data and 'Borough' is the column with borough names
  if 'borough' not in df.columns or 'average_speed' not in df.columns:
    print("Error: DataFrame must contain 'Borough' and 'Speed' columns.")
    return None

  average_speed = df.groupby('borough')['average_speed'].mean()
  return average_speed

def plot_average_speed_by_borough_colored(df):
  """
  Creates a bar graph showing the average speed per borough with different colors for each borough,
  float point labels on the y axis with 0.5 increments.

  Args:
    df: A pandas DataFrame containing bus speed data with a 'Borough' column
        and a column representing speed (e.g., 'Speed').
  """
  average_speed_by_borough_df = avg_speed_by_borough(df)
  if average_speed_by_borough_df is not None:
    plt.figure(figsize=(12, 7))
    ax = average_speed_by_borough_df.plot(kind='bar', color=sns.color_palette("viridis", len(average_speed_by_borough_df)))

    plt.title('Average Bus Speed per Borough', fontsize=16)
    plt.xlabel('Borough', fontsize=12)
    plt.ylabel('Average Speed (mph)', fontsize=12)
    plt.xticks(rotation=45, ha='right') # Rotate x-axis labels for better readability

    # Set y-axis limits and ticks for 0.5 increments
    y_max = average_speed_by_borough_df.max()
    plt.ylim(0, y_max + 1) # Add a small buffer
    plt.yticks(np.arange(0, y_max + 1, 0.5))
    ax.yaxis.set_major_formatter(plt.FormatStrFormatter('%.1f')) # Format as float

    plt.tight_layout() # Adjust layout to prevent labels overlapping
    plt.show()

def scatterplot_average_speed_by_borough_colored(df):
  """
  Creates a scatterplot showing the average speed per borough using different colors for each borough
   and starting the y-axis at 0.

  Args:
    df: A pandas DataFrame containing bus speed data with a 'Borough' column
        and a column representing speed (e.g., 'Speed').
  """
  average_speed_by_borough_df = avg_speed_by_borough(df)
  if average_speed_by_borough_df is not None:
    plt.figure(figsize=(12, 7))
    # Use seaborn scatterplot for easy coloring by category
    sns.scatterplot(x=average_speed_by_borough_df.index, y=average_speed_by_borough_df.values,
                    hue=average_speed_by_borough_df.index, s=100) # s is marker size

    plt.title('Average Bus Speed per Borough (Scatterplot)', fontsize=16)
    plt.xlabel('Borough', fontsize=12)
    plt.ylabel('Average Speed (mph)', fontsize=12)
    plt.xticks(rotation=45, ha='right') # Rotate x-axis labels for better readability

    # Start the y-axis at 0
    plt.ylim(0, average_speed_by_borough_df.max() + 1) # Add a small buffer

    plt.tight_layout() # Adjust layout to prevent labels overlapping
    plt.show()
